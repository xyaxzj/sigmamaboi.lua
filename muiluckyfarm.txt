local Players = game:GetService("Players")
local lp = Players.LocalPlayer

_G.autoUseBarbell = true

local targetTools = {
    ["Emerald Barbell"] = true,
    ["Planet Barbell"] = true
}

task.spawn(function()
    while task.wait(0.1) do
        if not _G.autoUseBarbell then continue end
        
        local char = lp.Character
        local hum = char and char:FindFirstChildOfClass("Humanoid")
        local backpack = lp:FindFirstChild("Backpack")
        
        if not char or not hum then continue end

        local currentTool = char:FindFirstChildOfClass("Tool")
        
        if currentTool and targetTools[currentTool.Name] then
            currentTool:Activate()
        else
            if backpack then
                for _, tool in ipairs(backpack:GetChildren()) do
                    if tool:IsA("Tool") and targetTools[tool.Name] then
                        hum:EquipTool(tool)
                        task.wait(0.1)
                        tool:Activate()
                        break
                    end
                end
            end
        end
    end
end)
